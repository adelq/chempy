[docs]def test_ArithmeticDict():
    d1 = ArithmeticDict(float, [('a', 1.0), ('b', 2.0)])
    d2 = ArithmeticDict(float, [('c', 5.0), ('b', 3.0)])
    d3 = d1+d2
    assert d3['a'] == 1.0
    assert d3['b'] == 5.0
    assert d3['c'] == 5.0
    d3 += {'c': 1.0}
    assert d3['a'] == 1.0
    assert d3['b'] == 5.0
    assert d3['c'] == 6.0
    d4 = {'a': 7.0} + d1
    assert d4['a'] == 8.0
    d5 = d1 + {'a': 9.0}
    assert d5['a'] == 10.0

    d6 = d1 - d2
    assert d6 == {'a': 1.0, 'b': -1.0, 'c': -5.0}
    d6 -= d3
    assert d6 == {'b': -6.0, 'c': -11.0}

[docs]def test_ArithmeticDict_div():
    d1 = ArithmeticDict(int, [('a', 6), ('b', 9)])

    d2 = d1/3
    assert d2['a'] == 2
    assert d2['b'] == 3

    d2 = d1.copy() / 3
    assert d2['a'] == 2
    assert d2['b'] == 3

    d2 = 54/d1
    assert d2['a'] == 9
    assert d2['b'] == 6
    assert d2['c'] == 0

    d3 = (6*d2)/d1
    assert d3['a'] == 9
    assert d3['b'] == 4
    assert d3['c'] == 0
